301 void ProSHADE_internal_io::readInMapData ( CMap_io::CMMFile* mapFile, proshade_double*& map, proshade_signed xIndexStart, proshade_signed yIndexStart, proshade_signed zIndexStart, proshade_signed xDimInIndices, proshade_signed yDimInIndices, proshade_signed zDimInIndices, proshade_signed xAxisOrder, proshade_signed yAxisOrder, proshade_signed zAxisOrder )
302 {
303  //================================================ Allocate map memory
304  map = new proshade_double [xDimInIndices * yDimInIndices * zDimInIndices];
305 
306  //================================================ Check memory allocation
307  ProSHADE_internal_misc::checkMemoryAllocation ( map, __FILE__, __LINE__, __func__ );
308 
309  //================================================ Check the map mode
310  int mapMode = CMap_io::ccp4_cmap_get_datamode ( mapFile );
311  if ( ( mapMode != 0 ) && ( mapMode != 2 ) )
312  {
313  throw ProSHADE_exception ( "The map file mode is not supported.", "EM00009", __FILE__, __LINE__, __func__, "ProSHADE currently only supports map modes 0 and 2. The\n : supplied map seems to have different map mode and\n : therefore cannot be read in." );
314  }
315 
316  //================================================ Read in map data
317  // ... Prepare the variables
318  proshade_signed index;
319  proshade_signed iters[3];
320  proshade_signed maxLim[3];
321  proshade_signed XYZOrder[3];
322  proshade_signed newU, newV, newW;
323  proshade_signed arrPos;
324  proshade_signed maxMapV = 0, maxMapW = 0;
325  proshade_signed order[3]; order[0] = xAxisOrder; order[1] = yAxisOrder; order[2] = zAxisOrder;
326  proshade_signed orig[3]; orig[0] = xIndexStart; orig[1] = yIndexStart; orig[2] = zIndexStart;
327  proshade_signed dim[3]; dim[0] = xDimInIndices; dim[1] = yDimInIndices; dim[2] = zDimInIndices;
328 
329  // ... Set the dimensions for XYZ indexing
330  if ( order[1] == 1 ) { maxMapV = dim[0] - 1; }
331  if ( order[1] == 2 ) { maxMapV = dim[1] - 1; }
332  if ( order[1] == 3 ) { maxMapV = dim[2] - 1; }
333 
334  if ( order[2] == 1 ) { maxMapW = dim[0] - 1; }
335  if ( order[2] == 2 ) { maxMapW = dim[1] - 1; }
336  if ( order[2] == 3 ) { maxMapW = dim[2] - 1; }
337 
338  // ... Set the XYZ indexing order and indices
339  for ( proshade_signed iter = 0; iter < 3; iter++ )
340  {
341  maxLim[iter] = orig[iter] + dim[iter] - 1;
342  XYZOrder[order[iter]-1] = iter;
343  }
344 
345  // ... Solve the dimensions and sizes for reading
346  proshade_signed fastDimSize = ( maxLim[0] - orig[0] + 1 );
347  proshade_signed midDimSize = ( maxLim[1] - orig[1] + 1 ) * fastDimSize;
348  std::vector < float > section ( midDimSize );
349 
350  // ... Read in the map data
351  for ( iters[2] = orig[2]; iters[2] <= maxLim[2]; iters[2]++ )
352  {
353  index = 0;
354  CMap_io::ccp4_cmap_read_section( mapFile, &section[0] );
355 
356  if ( mapMode == 0 )
357  {
358  for ( int iter = ( midDimSize - 1 ); iter >= 0; iter-- )
359  {
360  section[iter] = static_cast < float > ( ( reinterpret_cast<unsigned char*> (&section[0]) )[iter] );
361  }
362  }
363 
364  for ( iters[1] = orig[1]; iters[1] <= maxLim[1]; iters[1]++ )
365  {
366  for ( iters[0] = orig[0]; iters[0] <= maxLim[0]; iters[0]++ )
367  {
368  newU = iters[XYZOrder[0]] - orig[XYZOrder[0]];
369  newV = iters[XYZOrder[1]] - orig[XYZOrder[1]];
370  newW = iters[XYZOrder[2]] - orig[XYZOrder[2]];
371  arrPos = newW + ( maxMapW + 1) * ( newV + ( maxMapV + 1) * newU );
372  map[arrPos] = static_cast < proshade_double > ( section[ index++ ] );
373  }
374  }
375  }
376 
377  //================================================ Done
378  return ;
379 
380 }
