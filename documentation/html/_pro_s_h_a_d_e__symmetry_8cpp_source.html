528 bool ProSHADE_internal_symmetry::determineFoldToTry ( proshade_double dist, proshade_double* divBasis, proshade_double* divRem, proshade_double peakErr, proshade_double* symmErr, std::vector< proshade_unsign >* angsToTry )
529 {
530  //================================================ Initialise variables
531  bool ret = false;
532 
533  //================================================ Find the basis and remainder of the 2pi/dist equation
534  *divRem = std::modf ( static_cast<double> ( ( 2.0 * M_PI ) / std::abs ( dist ) ), divBasis );
535 
536  //================================================ If the remainder would be smaller for larger basis, so this basis
537  if ( *divRem > 0.5 )
538  {
539  *divRem -= 1.0;
540  *divBasis += 1.0;
541  }
542 
543  //================================================ Determine errors on peaks and on folds
544  *symmErr = ( M_PI * 2.0 / *divBasis ) - ( M_PI * 2.0 / ( *divBasis + 1.0 ) );
545  proshade_double angTolerance = ( peakErr / *symmErr );
546 
547  //================================================ Is remainder small enough?
548  if ( ( *divRem < ( 0.0 + angTolerance ) ) && ( *divRem > ( 0.0 - angTolerance ) ) )
549  {
550  //============================================ Are we sure about the fold determination accuracy
551  proshade_signed angTolRound = std::min ( ProSHADE_internal_mapManip::myRound ( angTolerance ), static_cast<proshade_signed> ( 10 ) );
552  for ( proshade_signed iter = -angTolRound; iter <= angTolRound; iter++ )
553  {
554  ProSHADE_internal_misc::addToUnsignVector ( angsToTry, static_cast<proshade_unsign> ( std::max ( *divBasis + iter, 2.0 ) ) );
555  }
556  }
557

559  if ( angsToTry->size() == 0 ) { ret = false; }
560  else { ret = true; }
561

696  for ( proshade_unsign xIt = 0; xIt < ( dataObj->getMaxBand() * 2 ); xIt++ )

699  {

704  pointHeight = pow( dataObj->getInvSO3Coeffs()[arrIndex][0], 2.0 ) + pow( dataObj->getInvSO3Coeffs()[arrIndex][1], 2.0 );

716  {

723  yPk *= -1.0;
724  zPk *= -1.0;

726  }
727

730  ( ( ( yPk - axTol ) < y ) && ( ( yPk + axTol ) > y ) ) &&
731  ( ( ( zPk - axTol ) < z ) && ( ( zPk + axTol ) > z ) ) )
732  {
733  if ( ret < pointHeight ) { ret = pointHeight; }
734  }
735  }
736  }
737  }
738  }
