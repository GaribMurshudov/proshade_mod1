375 void ProSHADE_internal_overlay::combineFourierForTranslation ( fftw_complex* tmpOut1, fftw_complex* tmpOut2, fftw_complex*& resOut, proshade_unsign xD, proshade_unsign yD, proshade_unsign zD )

380  proshade_signed uo2 = xD / 2;
381  proshade_signed vo2 = yD / 2;
382  proshade_signed wo2 = zD / 2;
383 
384  //================================================ Combine the coefficients
385  for ( proshade_signed uIt = 0; uIt < static_cast<proshade_signed> ( xD ); uIt++ )

394  if ( wIt > wo2 ) { l1 = wIt - zD; } else { l1 = wIt; }
395

398  if ( k1 < 0 ) { k1 += yD; }
399  if ( l1 < 0 ) { l1 += zD; }
400 
401  //==================================== Find indices
402  hlpPos = l1 + zD * ( k1 + yD * h1 );
403  arrPos = wIt + zD * ( vIt + yD * uIt );
404 
405  //==================================== Combine
406  ProSHADE_internal_maths::complexMultiplicationConjug ( &tmpOut1[hlpPos][0],
407  &tmpOut1[hlpPos][1],

409  &tmpOut2[arrPos][1],
410  &resOut[hlpPos][0],
411  &resOut[hlpPos][1] );
412 
413  //==================================== Save
